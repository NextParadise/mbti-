function showResult() { try { // 1. 隐藏答题区和进度条 document.getElementById('quizScreen').classList.remove('active'); document.getElementById('progressContainer').style.display = 'none'; // 2. 【关键修复】先让结果容器显示出来，防止隐藏状态下赋值报错 document.getElementById('resultScreen').classList.add('active'); // 3. 计算最终人格 let finalType = ""; finalType += scores['E'] >= scores['I'] ? "E" : "I"; finalType += scores['S'] >= scores['N'] ? "S" : "N"; finalType += scores['T'] >= scores['F'] ? "T" : "F"; finalType += scores['J'] >= scores['P'] ? "J" : "P"; // 4. 获取解析数据（增加容错 fallback） const resultData = typeDescriptions[finalType] || { title: "未知", desc: "计算出现偏差，请重试" }; // 5. 使用更稳定的 innerHTML 渲染数据 document.getElementById('resultType').innerHTML = finalType; document.getElementById('resultTitle').innerHTML = resultData.title; document.getElementById('resultDesc').innerHTML = resultData.desc; } catch (error) { // 即使发生极端错误，也能保证用户看到提示而不是白屏 console.error("渲染结果时出错:", error); document.getElementById('resultScreen').classList.add('active'); document.getElementById('resultDesc').innerHTML = "网络或浏览器状态异常，请点击下方按钮重新测试。"; } }
